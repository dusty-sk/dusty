Day/Night Cycle Scripts (not working)

Script 1

let a = 0, b = 0, c = 12 * 6e5, d = 24e3, e = d / c, f = 0, g = 0;

tick = ms => {
  a = (a + 0.002 * e) % (2 * Math.PI);
  b = (b + 0.005 * e) % Math.PI;
  const p = api.getPlayerIds(),
    h = ((f % c) * e) % d,
    i = ((ms % c) * e) % d;
  if (i < h) g++;
  f = ms;
  let j = i;
  if (g % 100 === 0 && i < 50) j = (j + 20) % d;

  const s0 = 23000, s1 = 4000, d0 = s1, d1 = 12000,
    s2 = d1, s3 = 13000, n0 = s3, n1 = s0;
  const r = j >= s0 || j < s1,
    q = j >= d0 && j < d1,
    t = j >= s2 && j < s3,
    u = !r && !q && !t;

  let v = [255, 225, 255], w = 1.8;

  if (u) {
    v = [128, 128, 179];
    w = 1.0;
  } else if (r) {
    const x = j >= s0
      ? (j - s0) / (d - s0 + s1)
      : (j + (d - s0)) / (d - s0 + s1);
    v = [Math.floor(25 + 230 * x), Math.floor(38 + 217 * x), Math.floor(128 + 127 * x)];
    w = 3.0 - 2.2 * x;
  } else if (t) {
    const x = (j - s2) / (s3 - s2);
    v = [Math.floor(255 - 230 * x), Math.floor(255 - 217 * x), Math.floor(255 - 127 * x)];
    w = 15.8 + 2.2 * x;
  } else if (q) {
    v = [255, 255, 255];
    w = 0.01;
  }

  for (const y of p) {
    api.setClientOption(y, "skyBox", {
      type: "earth",
      inclination: b,
      azimuth: a,
      turbidity: w,
      vertexTint: v
    });
  }
};
